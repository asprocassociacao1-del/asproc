📬 Para receber as mensagens do formulário de contato, crie também a tabela abaixo no SQL Editor do Supabase:

-- Tabela de mensagens de contato (sobre.html)
CREATE TABLE IF NOT EXISTS mensagens_contato (
  id          bigserial PRIMARY KEY,
  nome        text NOT NULL,
  email       text NOT NULL,
  assunto     text,
  mensagem    text NOT NULL,
  created_at  timestamptz DEFAULT now()
);

-- Permissão pública para inserir (anon pode enviar mensagem)
ALTER TABLE mensagens_contato ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Inserir mensagens" ON mensagens_contato
  FOR INSERT TO anon WITH CHECK (true);

CREATE POLICY "Admin lê mensagens" ON mensagens_contato
  FOR SELECT TO authenticated USING (true);

CREATE POLICY "Admin exclui mensagens" ON mensagens_contato
  FOR DELETE TO authenticated USING (true);

          🛠 SQL de criação da tabela (executar
            1x no Supabase)
          
-- Execute este SQL no Supabase → SQL Editor:

CREATE TABLE IF NOT EXISTS publicacoes_agendadas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  tabela text NOT NULL CHECK (tabela IN ('noticias','vagas','avisos')),
  registro_id text NOT NULL,
  titulo text,
  publicar_em timestamptz NOT NULL,
  status text NOT NULL DEFAULT 'pendente' CHECK (status IN ('pendente','publicado','cancelado','erro')),
  criado_por text,
  criado_em timestamptz DEFAULT now(),
  publicado_em timestamptz,
  erro_msg text
);

ALTER TABLE publicacoes_agendadas ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_agendamentos" ON publicacoes_agendadas
  FOR ALL TO authenticated USING (true) WITH CHECK (true);

-- Índice para o job de verificação ser rápido:
CREATE INDEX IF NOT EXISTS idx_agend_status_pub
  ON publicacoes_agendadas(status, publicar_em)
  WHERE status = 'pendente';
